// 自定义一个Person类
public class Person
{
    // 字段
    private string name;
    private int age;

    // 属性
    public string Name
    {
        get { return name; }
        set { name = value; }
    }

    public int Age
    {
        get { return age; }
        set { age = value; }
    }

    // 方法
    public virtual void SayHello()
    {
        Console.WriteLine("Hello, I am a person.");
    }
}

// 自定义一个Student类继承自Person类
public class Student : Person
{
    // 字段
    private int grade;

    // 属性
    public int Grade
    {
        get { return grade; }
        set { grade = value; }
    }

    // 方法
    public override void SayHello()
    {
        Console.WriteLine("Hello, I am a student.");
    }

    public void Study()
    {
        Console.WriteLine("I am studying.");
    }
}

// 自定义一个Dog类
public class Dog
{
    // 字段
    private string name;
    private int age;

    // 属性
    public string Name
    {
        get { return name; }
        set { name = value; }
    }

    public int Age
    {
        get { return age; }
        set { age = value; }
    }

    // 方法
    public void Bark()
    {
        Console.WriteLine("Woof woof!");
    }
}

// 自定义一个Poodle类继承自Dog类
public class Poodle : Dog
{
    // 字段
    private string haircut;

    // 属性
    public string Haircut
    {
        get { return haircut; }
        set { haircut = value; }
    }

    // 方法
    public void ShowOffHaircut()
    {
        Console.WriteLine("I have a nice haircut: " + haircut);
    }
}
